% A command that places prompts in a breakable box
\newcommand{\renderprompt}[3]{%
  \begin{figure}[t]
  \begin{adjustbox}{max width=\textwidth}
  \begin{promptbox}
  \VerbatimInput[numbers=left,
                 numbersep=5pt,
                 frame=none,
                 fontsize=\scriptsize,
                 commandchars=\\\{\},
                 breaklines=true,
                 breakanywhere=true,
                 breakindent=0pt,
                 breakautoindent=false,
                 breaksymbol={},
                 breaksymbolindent=0pt,
                 breaksymbolsepleft=0pt,
                 breaksymbolsepright=0pt]
                {#1}
  \end{promptbox}
  \end{adjustbox}
  \caption{#3}
  \label{#2}
  \end{figure}
}
